\section{机器人末端位姿描述}

\begin{quote}
    Kinetic Position and Coordinate
\end{quote}

机械手的位置表达：
\begin{enumerate}
    \item 机械手末端姿态由某个姿态变换规定
    \item 相对基系的位置由\textbf{姿态矩阵}由左乘对应于矢量\(p\)的平移变换确定
\end{enumerate}

\subsection{Description in Cylinder Coordinates}

用柱面坐标表示末端位置（平移变换）

规定运动变换的过程(相对固定坐标系$xyz$运动)：
\begin{enumerate}
    \item 沿$x$轴平移$r$
    \item 绕$z$轴旋转$\alpha$
    \item 沿\(z\)轴平移\(z\)
\end{enumerate}

$$
\begin{tikzpicture}[
    % 设置 3D 透视坐标系 (Cavalier projection)

]
    % 定义主要坐标点

    \draw[->, thick, >=latex] (O) -- (6,0,0) node[right] {$x$};
    \draw[->, thick, >=latex] (O) -- (0,5.5,0) node[below] {$y$};
    \draw[->, thick, >=latex] (O) -- (0,0,6) node[left] {$z$};

    % 2. 绘制平面投影线及辅助线
    \draw (O) -- (Pxy); % 连向 xy 平面投影点的斜线 (r)
    \draw (Px) -- (Pxy); % 平行于 y 轴的辅助线
    \draw (Pxy) -- (P) node[midway, right] {$z$}; % 垂直于 xy 平面的高度线

    % 3. 绘制主向量
    % 使用 shorten >= 缩短一点箭头，以还原原图中箭头与局部坐标系原点间的微小空隙
    \draw[->, thick, >=latex, shorten >= 2pt] (O) -- (P);

    % 4. 绘制 xy 平面内的角度弧线和标签
    \begin{scope}[canvas is xy plane at z=0]
        % 角度 \alpha 的圆弧 (从 x 轴到投影线)
        \draw[->, >=latex] (1.5, 0) arc (0:53.13:1.5);
        \node at (26.5:1.1) {$\alpha$};
        % 标签 r 放置在靠近 x 轴的外侧
        \node at (10:2.1) {$r$};
    \end{scope}

    % 5. 绘制点 P 处的局部坐标系 (a, o, n)
    % 向量 n (指向右前/下方)
    \draw[->, thick, >=latex] (P) -- ++(2.0, 2.0, 0) node[right] {$\boldsymbol{n}$};
    % 向量 o (指向右后/上方)
    \draw[->, thick, >=latex] (P) -- ++(-1.5, 1.0, 0) node[right] {$\boldsymbol{o}$};
    % 向量 a (指向上方，平行于 z 轴)
    \draw[->, thick, >=latex] (P) -- ++(0, 0, 1.5) node[left] {$\boldsymbol{a}$};
\end{tikzpicture}
$$

变换算子：
$$
\operatorname{Cyl}(z, \alpha, r) = \operatorname{Trans}(0,0,z) \operatorname{Rot}(z, \alpha) \operatorname{Trans}(r,0,0)
$$

\begin{note}
    相对运动坐标$xyz$的变换算子：
    $$
    \operatorname{Cyl}(z, \alpha, r) = \operatorname{Trans}(0,0,z) \operatorname{Rot}(z^\prime, \alpha) \operatorname{Trans}(r^\prime,0,0)
    $$
\end{note}

\subsection{Description in Spherical Coordinates}

用球面坐标表示末端运动位置矢量

规定运动变换的过程(相对固定坐标系$xyz$运动)：
\begin{enumerate}
    \item 沿$z$轴平移$r$
    \item 绕$y$轴旋转$\beta$
    \item 沿\(z\)轴旋转\(\alpha\)
\end{enumerate}

变换算子：
$$
\operatorname{Sph}(z, \alpha, r) = \operatorname{Rot}(z, \alpha) \operatorname{Rot}(y,\beta) \operatorname{Trans}(0,0,r)
$$

\begin{note}
    相对运动坐标$xyz$的变换算子：
    $$
    \operatorname{Sph}(z, \alpha, r) = \operatorname{Rot}(z^\prime, \alpha) \operatorname{Rot}(y^\prime,\beta) \operatorname{Trans}(0,0,r^\prime)
    $$
\end{note}

\begin{itemize}
    \item 杆件编号由固定基座开始，一般称基座为连杆0
    \item 关节1处于连杆1和基座之间，杆件距基座近的一端的关节为第$i$个关节
    \item 坐标系0是一个固定不动的坐标系（参考坐标系）
    \item $n$关节机器人需建立$n+1$个坐标系
\end{itemize}

\subsection{连杆参数}

两个结构参数：
\begin{itemize}
    \item 长度 \( a_{i-1} \)：第 \(i-1\) 轴指向第 \(i\) 轴的公法线长度（恒为正）。
    \item 扭角 \( \alpha_{i-1} \)：第 \(i-1\) 轴绕公垂线转至第 \(i\) 轴的夹角（可正可负）。
\end{itemize}

两个连接参数：
\begin{itemize}
    \item 偏置 \( d_i \)：两条公法线之间的距离（带正负号）。
    \item 关节角 \( \theta_i \)：两条公法线之间的夹角（带正负号）。
\end{itemize}

连杆 \(i-1\) 的四个参数为：\(a_{i-1}\)、\(\alpha_{i-1}\)、\(d_i\)、\(\theta_i\)。

\begin{note}
    \begin{itemize}
        \item \textbf{旋转关节 (Revolute Joint)}：关节变量 \(\theta_i\)，连杆参数 \(a_{i-1}\)、\(\alpha_{i-1}\)、\(d_i\) 不变。
        \item \textbf{移动关节 (Prismatic Joint)}：关节变量 \(d_i\)，连杆参数 \(a_{i-1}\)、\(\alpha_{i-1}\)、\(\theta_i\) 不变。
    \end{itemize}
\end{note}

\dfn{Denavit—Hartenberg notation}{Hence, any robot can be described kinematically by giving the values of four
quantities for each link. Two describe the link itself, and two describe the link's
connection to a neighboring link. In the usual case of a revolute joint, is called
the joint variable, and the other three quantities would be fixed link parameters.
For prismatic joints, d1 is the joint variable, and the other three quantities are fixed
link parameters. The definition of mechanisms by means of these quantities is a
convention usually called the Denavit—Hartenberg notation}

\ex{连杆坐标系建立的步骤}{
    \begin{enumerate}
        \item 找出各个关节轴
        \item 画出相邻关节轴线的公垂线
        \item 规定$z_i$轴与关节轴$i$重合
        \item 规定$x_i$轴与公垂线$a_i$重合
        \item 定义$y_i = z_i \times x_i$
        \item 按规则确定$\{0\}$、$\{n\}$
    \end{enumerate}
}
